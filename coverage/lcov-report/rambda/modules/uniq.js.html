const contains = require("./contains")
 
function uniq(arr){
  let index = -1
  const willReturn = []
  while (++index < arr.length) {
    const value = arr[ index ]
    if (!contains(value, willReturn)) {
      willReturn.push(value)
    }
  }
 
  return willReturn
}
 
module.exports = uniq
